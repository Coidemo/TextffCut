## 最も実用的な実装案

### **ハイブリッドアプローチ**

```python
def render_hybrid_timeline_editor():
    """テキストと視覚要素を組み合わせた編集UI"""
    
    # 1. 動画プレーヤー（オプション）
    if st.checkbox("プレビューを表示"):
        st.video(video_path)
    
    # 2. セグメントリスト（メイン）
    for i, segment in enumerate(segments):
        with st.container():
            col1, col2, col3 = st.columns([1, 4, 1])
            
            with col1:
                # 選択ボタン
                if st.button(f"#{i+1}", key=f"sel_{i}"):
                    select_segment(i)
            
            with col2:
                # テキストとタイムスタンプ
                st.markdown(f"**{format_time(segment.start)} - {format_time(segment.end)}**")
                st.text(segment.text[:100] + "...")
            
            with col3:
                # アクション
                if st.button("編集", key=f"edit_{i}"):
                    edit_segment(i)
    
    # 3. 選択中のセグメント編集
    if selected_segment:
        st.divider()
        render_segment_editor(selected_segment)
```
